def findSlopeIntercept(coefs,bias): 
    hidden_wts=coefs[0] 
    bias_wts=bias[0] 
    num=len(hidden_wts[0]) 
    slopIntercept=[]
    for i in range(num): 
        w=[bias_wts[i],hidden_wts[0,i],hidden_wts[1,i]] 
        slope=w[1]/w[2]*(-1)
        intercept=w[0]/w[2]*(-1) 
        slopIntercept.append([slope,intercept])
    return(slopIntercept)
